#### 1️⃣ 6485_삼성시의 버스노선

```python
T = int(input())
for tc in range(1,T+1) :
    # 노선 수
    N = int(input())

    # 노선 정보 받기
    inf_lst = []
    for _ in range(N) :
        inf_lst.append(list(map(int,input().split())))
    # print(inf_lst)

    # 정류장 갯수
    P = int(input())
    for _ in range(P) :
        int(input())
   
    ans_lst = [0] * (P+1)
    for i in inf_lst :
        a = i[0]
        b = i[1]
        while a <= b :
            ans_lst[a] += 1
            a += 1
    print(f'#{tc}', *ans_lst[1::])
```
